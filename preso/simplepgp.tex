\begin{itemize}
\item PGP, GnuPG, OpenPGP
\item Used for: 
\begin{itemize}
\item Digital Signature (DSS/SHA or RSA/SHA)
\item Message Encryption(CAST-128, IDEA, 3-DES with RSA)
\item Compression 
\item Compatibility (Radix-64 conversion)
\end{itemize}
\item 1998, IETF WG, RFC 4880
\item interoperability
\item legal issues
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{How it works}
\begin{enumerate}
\item 4 key types: passphrase, session-key, private key, public key
\item Public and private key rings
\item Passphrase encrypts private key ring
\item Every user can have a public-private key pair
\end{enumerate}
\end{frame}
